## Setup pom.xml
```
<repositories>
   <repository>
      <id>jacob</id>
      <url>https://raw.github.com/yipuran/jacob/mvn-repo</url>
   </repository>
</repositories>

<dependencies>
    <dependency>
        <groupId>org.jacob</groupId>
        <artifactId>jacob</artifactId>
        <version>2.0</version>
    </dependency>
</dependencies>

```
